    # 1、教材的算法3.1。对于输入的任意一个非负十进制整数，打印输出与其等值的八进制数。

    #include<iostream>
using namespace std;

//定义链栈结构体 
struct stack{
	int data;
	stack* next;
}; 

stack* get(int n){
	stack *H=NULL;				//初始化链栈 
	while(n!=0){				//除8取余法求八进制 
		//链栈插入操作 
		stack *p=new(stack);
		p->data=n%8;
		p->next=H;				//关键步骤 
		H=p;					//关键步骤 
		n/=8;
	}
	return H;	
}

//链栈出栈操作 
void show(stack *p){
	while(p!=NULL){				//判空条件 
		cout<<p->data;
		p=p->next;	
	}
}

int main(){
	int n;
	cout<<"请输入要转换的十进制数：";
	cin>>n;
	cout<<"八进制数为：";
	show(get(n));
	return 0;
}

    # 2、汉诺塔问题。用户输入盘的数量N，程序输出移动盘的过程。

    # 用到递归方法，虽用户未定义栈，但系统也会自动调用栈

    #include<iostream>
using namespace std;
int num;

void move(int n,char x,char z){
	cout<<"第"<<++num<<"次移动："<<n<<"盘从"<<x<<"塔移动到"<<z<<"塔"<<endl;
}

void hanoi(int n,char x,char y,char z){
	if(n==1){
		move(1,x,y,z);				//将最后一个盘从x塔移动到z塔，借助y塔 
	}else{
		hanoi(n-1,x,z,y);			//将1至n-1个盘从x塔移动到y塔，借助z塔 
		move(n,x,z);				//将n盘从x塔移动到z塔 
		hanoi(n-1,y,x,z);			//将1至n-1个盘从y塔移动到z塔，借助x塔 
	}
}

int main(){
	char x='A',y='B',z='C';
	cout<<"请输入圆盘个数：";
	int n;
	cin>>n;
	hanoi(n,x,y,z);
	return 0;
}
