    # 1、教材的算法4.1。T为非空串。若主串S中第pos个字符之后存在与T相等的子串，则返回第一个这样的子串在S中的位置，否则返回0。

    #include<iostream>
using namespace std;

int bf(string s,string sub,int pos){
	pos-=1;											//string中字符串下标从0开始
	int i,j=0;
	for(i=pos;i<=s.length()-sub.length();){			//从主串的第pos位开始查找，到最后一个可能位置结束 
		for(j=0;j<sub.length();){
			if(s[i++]==sub[j++]){					//子串与主串进行匹配 
			}else{
				i=i-j+1;							//不匹配则回溯并跳出本次循环 
				break; 
			}
			if(j==sub.length()) return i-j;			//若子串完整匹配完，结束所有循环并返回主串位置（注意j是从0开始算） 
		}
	}
	return -1;										//所有循环结束，未能匹配成功 
}

int main(){
	string s,sub;
	int pos,p;
	cout<<"请输入主串：";
	cin>>s;
	cout<<"请输入开始查找的位置：";
	cin>>pos;
	cout<<"请输入子串：";
	cin>>sub;
	if(s.length()<sub.length()){
		cout<<"子串过长！";
		return 0;
	}
	if(pos<=0||pos-1>s.length()-sub.length()){		//判断起始查找位置合法性 
		cout<<"查找位置错误！";
		return 0;
	}
	p=bf(s,sub,pos);								//BF串匹配算法，返回主串中符合要求的位置 
	if(p>=0){										//失败返回-1，成功则返回主串位置 
		cout<<"子串位于主串"<<p+1<<"的位置";
	}else{
		cout<<"未能匹配";
	};
	return 0;
}

    # 2、教材的算法4.3。用Sub返回串S的第pos个字符起长度为len的子串。

    #include<iostream>
using namespace std;

int getSub(string s,int pos,int len){
	if(pos+len-1>s.length()) return 0;		//判断起始条件是否合法 
	string sub;
	pos-=1;									//字符串下标从0开始 
	for(int i=0;i<len;i++){					//获取字串 
		sub[i]=s[pos+i];
	}
	for(int j=0;j<len;j++){					//输出字串 
		cout<<sub[j];	
	}
}

int main(){
	string s,sub;
	int pos,len;
	cout<<"请输入要查找的主串：";
	cin>>s;
	cout<<"请输入查找位置：";
	cin>>pos;
	cout<<"请输入查找的子串长度：";
	cin>>len;
	if(!getSub(s,pos,len)) cout<<"请输入正确的查找范围！";
	return 0;
}
