    #include<algorithm>				//用于调用sort()函数的头文件 
#include<iostream>
using namespace std;

//非递归实现的折半查找算法 
int search(int* l,int low,int high){
	if(low>high) return 0;		//查找范围错误返回0
	while(low<=high){			//注意一定要带上等号 
		int mid=(low+high)/2;
		if(l[0]==l[mid]) return mid;
		else if(l[0]<l[mid]) high=mid-1;
		else low=mid+1;
	}
	return 0;					//如果在循环结束都没有退出函数，说明没找到该数，返回0 
}

	cin>>l[0]; 					//将待查找的数存为哨兵，折半查找中没啥意义，可以不用

	if(!n) cout<<"查无此数！";
	else cout<<endl<<"该数按从小到大顺序在数列中排第"<<n<<"位"; 
	return 0;
}

    # 递归实现的折半查找完整代码

    #include<algorithm>				//用于调用sort()函数的头文件 
#include<iostream>
using namespace std;

//递归实现的折半查找算法 
void search(int* l,int low,int high){
	if(low>high) cout<<"查无此数";				     	//递归结束条件 
	else{
		int mid=(low+high)/2;
		if(l[mid]==l[0]) cout<<"该数存在！";			//每次都判断中点的值是否为查找值 
		else if(l[0]<l[mid]) search(l,low,mid-1);		//分治递归 
		else search(l,mid+1,high); 
	}
}

int main(){
	int l[100],num,len;

	sort(l+1,l+len+1);			//用stl进行快速排序 
	search(l,1,len);
	return 0;
}
