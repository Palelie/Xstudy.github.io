    # 二、深度优先遍历（BFS）

    # 简而言之：一条路走到黑，然后一步步后退，寻找还未被访问的道路，然后继续走到黑，一直循环该步骤直到全部已经访问完成

    # 用DFS遍历由邻接矩阵表示的图

    # 步骤解析：

    1. 准备一个visited数组用来记录结点是否已经遍历过，0表示未遍历，1表示已遍历
    2. 准备好图的邻接矩阵，从起点的行开始遍历（设行为i，列为j）
    3. 遇到1时判断该点（j）是否已经遍历，若visited[j]=0，将0改为1，然后从邻接矩阵第j行开始遍历
    4. 不断地深入，直到没路可走，返回上一个结点，继续遍历
    5. 当visited所有值都为1时遍历结束

    # 用BFS遍历由邻接表表示的图

    # 步骤解析：

    1. 准备一个队列和一个visited数组
    2. 遍历链表结点时判断visited数组，若未访问过则入队，同时输出结点并将visited改1
    3. 当一个表头结点的所有链表结点都遍历完后，出队，然后将该结点作为下一个遍历的表头结点
    4. 队空时遍历结束

    # 1、递归实现的DFS算法

    #include<iostream>
#include<stack> 
using namespace std;
const int MAXSIZE = 10000; 

struct graph{
	char vex[10];					//存储结点 
	int arc[10][10];				//存储边 
	int vexnum;						//结点个数 
}; 

//创建邻接矩阵 
void creatGraph(graph &g,int n){
	g.vexnum=n;
	cout<<"input the vexs:";				//输入结点 
	for(int i=0;i<n;i++) cin>>g.vex[i];
	cout<<"input the graph:"<<endl;			//输入邻接矩阵 
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
		cin>>g.arc[i][j];
		}
	}
}

//获得结点在结点数组里的下标 
int locateVex(graph g,char ch){
	for(int i=0;i<g.vexnum;i++){
		if(ch==g.vex[i]) return i;
	}
}

//基于邻接矩阵的递归深度优先遍历算法 
//参数1：邻接矩阵；参数2：visited数组；参数3：开始遍历的第一个结点在结点数组里的下标 
void DFS(graph &g,int *visited,int v){		
	cout<<g.vex[v];							//步骤1：输出该结点 
	visited[v]=1;								//步骤2：修改visited数组 
	for(int i=0;i<g.vexnum;i++){			//步骤3：遍历该结点的行邻接矩阵
		if(!visited[i]&&g.arc[v][i]){			//若未遍历且邻接矩阵不为0，继续向下一个结点递归循环 
			DFS(g,visited,i);
		}
	}
}

	

int main(){
	graph g;
	char ch;
	int n,v,visited[MAXSIZE];
	cout<<"input the vexs numbers:"; 			//输入结点个数 
	cin>>n;
	creatGraph(g,n);
	for(int i=0;i<n;i++) visited[i]=0;		//初始化visit数组
	cout<<"input the start vex:";				//输入遍历的第一个结点 
	cin>>ch; 
	v=locateVex(g,ch);							//获取输入的结点在数组里的下标
	DFS(g,visited,v);
	return 0;
}

    # 2、非递归实现的DFS算法

    #include<iostream>
#include<stack> 
using namespace std;
const int MAXSIZE = 10000;

struct graph{
	char vex[10];
	int arc[10][10];
	int vexnum;
}; 

//创建邻接矩阵 
void creatGraph(graph &g,int n){
	g.vexnum=n;
	cout<<"input the vexs:";				//输入结点 
	for(int i=0;i<n;i++) cin>>g.vex[i];
	cout<<"input the graph:"<<endl;			//输入邻接矩阵 
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
		cin>>g.arc[i][j];
		}
	}
}

//获得结点在结点数组里的下标 
int locateVex(graph g,char ch){
	for(int i=0;i<g.vexnum;i++){
		if(ch==g.vex[i]) return i;
	}
}

//基于邻接矩阵的非递归深度优先遍历算法 
//参数1：邻接矩阵；参数2：visited数组；参数3：开始遍历的第一个结点在结点数组里的下标  
void DFS(graph &g,int* visited,int v){
	stack<int> s;							//创建栈容器以用来做递归调用 
	s.push(v);								//将第一个遍历的结点入栈 
	visited[v]=1;
	cout<<g.vex[v];							//入栈时输出 
	while(!s.empty()){						//栈空时表示遍历结束 
		for(int i=0;i<g.vexnum;i++){		//从结点v开始一条路走到底！！！！ 
			if(g.arc[v][i]&&!visited[i]){
				cout<<g.vex[i];
				visited[i]=1;
				s.push(i);					//将遍历的结点入栈 
				v=i; 						//下一次循环从入栈结点那一行开始 
				i=-1;						//从第一列开始重新遍历（i++后i=0） 
			}
		}
		s.pop();							//某条路走到底后，返回上一步（出栈），看看还有没有路 
		if(!s.empty()) v=s.top();	//如果还能够返回上一步，让v等于上一步的结点，继续一条路走到底 
	}
	
}

int main(){
	graph g;
	char ch;
	int v,n,visited[MAXSIZE];
	cout<<"input the vexs numbers:"; 			//输入结点个数 
	cin>>n;
	for(int i=0;i<n;i++) visited[i]=0;			//初始化visit数组
	creatGraph(g,n);
	cout<<"input the start vex:";				//输入遍历的第一个结点 
	cin>>ch; 
	v=locateVex(g,ch);	
	DFS(g,visited,v);
	return 0;
}

    # 3、BFS算法

    #include<iostream>
#include<queue>
using namespace std;
const int MAXSIZE = 10000;


struct graph{
	char vex[10];					//存储结点 
	int arc[10][10];				//存储边 
	int vexnum;						//结点个数 
}; 


//创建邻接矩阵 
void creatGraph(graph &g,int n){
	g.vexnum=n;
	cout<<"input the vexs:";				//输入结点 
	for(int i=0;i<n;i++) cin>>g.vex[i];
	cout<<"input the graph:"<<endl;			//输入邻接矩阵 
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
		cin>>g.arc[i][j];
		}
	}
}

//获得结点在结点数组里的下标 
int locateVex(graph g,char ch){
	for(int i=0;i<g.vexnum;i++){
		if(ch==g.vex[i]) return i;
	}
}


void BFS(graph g,int* visited,int v){
	queue<int> q;							//创建队列容器 
	int j=0;
	q.push(v);								//将第一个结点入队 
	visited[v]=1;
	cout<<g.vex[v];							//入队时输出 
	while(!q.empty()){						//队列为空时遍历结束 
		v=q.front();						//让v等于出队结点
		q.pop();
		for(int j=0;j<g.vexnum;j++){	//寻找与v相邻的结点，入队并输出（每次都完整遍历完一整行） 
			if(g.arc[v][j]&&!visited[j]){
				q.push(j);
				visited[j]=1;
				cout<<g.vex[j];
			}
		}
	}
}

int main(){
	graph g;
	int n,v,visited[MAXSIZE];
	char ch;
	cout<<"input the vexs numbers:"; 
	cin>>n;
	for(int i=0;i<n;i++) visited[i]=0;		//初始化visit数组 
	cout<<"input the start vex:";			//输入开始遍历的结点 
	cin>>ch;
	v=locateVex(g,ch);						//获取输入的结点在数组里的下标 
	creatGraph(g,n);
	BFS(g,visited,v);
	return 0;
}
