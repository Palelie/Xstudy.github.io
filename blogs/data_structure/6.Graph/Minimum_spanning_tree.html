    # 三、构造最小生成树

    # MST性质：（本质上是贪心算法思想）

    # 构造算法一：普里姆（Prim）算法

    1. 将图上的结点分为两个集合U和V-U（V是所有点的集合）
    2. 每次寻找连接U和V-U两个集合的最小权值的边，然后把该边连接的点从V-U集合移动到U集合中
    3. 当所有点都加到U集合（即U=V）结束

    # 构造算法二：克鲁斯卡尔（Kruskal）算法

    1. 先列出所有点（没有边）
    2. 每次从边集合中找到权值最小的边
    3. 判断该边是否会与已有的边形成回路，若不会则加入，若会则找下一条权值最小的边
    4. 结束条件：所有的顶点都在同一连通分量上（即加入的边的数量为n-1）

    #include<iostream>
using namespace std;

struct edgeC{						//存放边和对应的两个结点 
	int from,to;
	int wei;
};

struct graph{						//存放边、点、数量信息 
	edgeC edge[100];
	int verNum,edgeNum;
}gra;

//给储存权值的数组按权值大小排序（快速排序法） 
void weiSort(int l,int r){
	if(l<r){
		int i=l,j=r;	
		edgeC x=gra.edge[i];								//基准数取数组的第一位 
		while(i<j){											//挖坑+填坑 
			while(i<j && gra.edge[j].wei>=x.wei) j--;
			if(i<j) gra.edge[i++]=gra.edge[j];
			while(i<j && gra.edge[i].wei<=x.wei) i++;
			if(i<j) gra.edge[j--]=gra.edge[i];
		}
		gra.edge[i]=x;
		weiSort(l,i-1);										//递归分治 
		weiSort(i+1,r);
	}
}

int getpar(int g,int *par){			//寻找结点的最大父节点 
	while(par[g]!=-1) g=par[g];
	return g;
}

//克鲁斯卡尔算法 
void kruskal(int *par){
	int sum=0;
    
	//寻找构成最小生成树的n-1条边
	for(int i=0,j=0;i<gra.verNum-1&&j<gra.edgeNum;j++){		 
        //i为成功加入的边的个数，j为循环到第j条边 
        
		if(getpar(gra.edge[j].from,par)!=getpar(gra.edge[j].to,par)){		
            //当该边的两个结点的根（父）结点不同，该边加入 
            
			par[getpar(gra.edge[j].to,par)]=getpar(gra.edge[j].from,par);	
            //修改新加入的边的根（父）节点 
            
			sum+=gra.edge[j].wei;			//计算总值 
			cout<<"铁路"<<i+1<<":"<<gra.edge[j].from<<"—"<<gra.edge[j].to<<"	路程："<<gra.edge[j].wei<<"千米"<<endl;
			i++;							//成功加入一条边 
            
			if(j==gra.edgeNum) cout<<"该图不是连通图";		
            //当所有边已经找完了但符合要求的边还不到n-1条，说明不是连通图 
		} 
	}
	cout<<endl<<"总价格为："<<sum*1000<<"万元"; 
}

int main(){
	int par[100]; 						//parent数组用于记录各个结点的根（父）节点 
	//输入图的信息：结点和边信息
	cout<<"请输入结点和边的个数：";
	//参考数据：6 10 
	cin>>gra.verNum>>gra.edgeNum;
	cout<<"请输入结点和边的权值："<<endl;
//	参考数据：
//	1 0 6
//	3 5 2 
//	5 4 6
//	2 5 4
//	2 3 5
//	0 2 1
//	2 4 6
//	1 2 5
//	1 4 3
//	0 3 5
	for(int i=0;i<gra.edgeNum;i++)
	 	cin>>gra.edge[i].from>>gra.edge[i].to>>gra.edge[i].wei;
	weiSort(0,gra.edgeNum-1);									//将边的权值进行排序
	for(int i=0;i<gra.verNum;i++) par[i]=-1;					//初始化parent数组 
	kruskal(par);
	return 0;
}
