    #include<iostream>
using namespace std;

struct List{			//链表结点 
	int data;
	List* next;
}; 

 
void InitList(List*&L){
	L=new List;			//头指针指向一个空的头结点 
	L->next=NULL;
}

int main(){
	List *head;			//创建头指针
	InitList(head);		//头指针初始化 
	return 0;

    # 注意！

    尽量不要在函数中请求一块新的内存，容易造成内存泄漏（忘记delete）

    如果一定需要在函数中请求新内存，将该块内存传回主函数有两种方法：

    1. 函数参数使用*&形式

      void InitList(List*&L){		//注意参数类型！！！
	L=new List;			
	L->next=NULL;
}

int main(){
	List *head;			
	InitList(head);		
	return 0;
}

    # 总结：凡是函数参数的内存地址发生改变，若需传回改变后的地址均需要采用以上两种方法！

    #include<iostream>
using namespace std;

//无效操作
void xx(int *b){		//执行该函数，值为原地址指向的值 
	int i=2;
	b=&i;				//该操作不会改变main函数中b的地址 
}

//方法一
void yy(int*&b){		//执行该函数，值为3
	int i=3;
	b=&i;				//操作有效 
}

//方法二
int* zz(int *b){		//执行该函数，值为4
	int i=4;
	b=&i;
	return b;			//操作有效 
}

void oo(int *b){		//执行该函数，值为5
	int i=5;
	*b=i;				//地址不变，标准的指针用法 
}

int main(){
	int a=1;
	int *b=&a;
	xx(b);			//执行该函数，值为 1
	yy(b);			//执行该函数，值为3
	b=zz(b);		//执行该函数，值为4
	oo(b);			//执行该函数，值为5
	cout<<*b;
}

    //销毁单链表
//头结点的内存空间也会被释放
void DestroyList(List*&L){	   	//注意delete同new一样，参数需要用*&的形式！！！
	List *p;			 
	while(L){
		p=L;
		L=L->next;
		delete p;			//c语言为free()
	}
}

    # 四、清空单链表

    # 注意：置空后链表（头结点和头指针）仍存在

    # 写法一：

    //清空单链表 
void ClearList(List *L){
	List *p;
	p=L->next;				//注意此时p指向的是首元结点而不是头结点！ 
	L->next=NULL;			//将头结点的next指针域置空 
	while(p){				//没到表尾 
		L=p->next;
		delete p;
		p=L;
	}
}

    # 五、求表长

    //求表长
int ListLength(List *L){
	int num=0;
	List *p;
	p=L->next;			//p指向首元结点 
	while(p){
		p=p->next;
		num++;
	}	
	return num;
}

    # 六、单链表的查找

    # 1、取单链表中第i个元素的内容

    # 时间复杂度O(n)

    //取值——取单链表中第i个元素的内容
int GetElem(List *L,int i){
	int j=1;							//j为计数器 
	L=L->next;							//指针指向首元结点 
	while(L && j<i){					//L不为空且计数器j还没等于i时，循环继续 
		L=L->next;	
		j++;
	}
	if(L&&j==i) return L->data;			//当j=i且L不为空指针时返回第i个元素的值 
	else cout<<"第"<<i<<"个元素不存在！"; 
}

    # 2、按值查找——根据特定数据获取该数据所在地址

    # 时间复杂度O(n)

    //按值查找——根据指定数据获取该数据位置序号
int LocateElem_L(List *L,int e){
	int j=1;				//计数器 
	L=L->next;
	while(L&&L->data!=e){	 
		L=L->next;
		j++;
	}
	if(L) return j;			//结点非空返回计数 
	else return 0;			//结点为空，证明遍历到最后都找不到该数据，返回0 
}

    # 七、单链表结点的插入

    # 时间复杂度：O(n)

    //在L中第i个位置之前插入新的数据元素e,L的长度+1
//前提:1<=i<=ListLength(L)；但即使i非法在函数中也会识别
void ListInsert(List *L,int i,int e){
	int j=0;					//计数器
	 while(L&&j<i-1){			//找到第i-1个结点(若为双向链表找到第i个结点也可操作) 
		L=L->next;
		j++;
	 }
	 if(L&&j==i-1){				//L非空且计数器j=i-1 
	 	List *p=new List;		//生成新结点 
	 	p->data=e;
	 	p->next=L->next;		//关键步骤 
	 	L->next=p;				//关键步骤 
	 } 
	 else cout<<"插入位置非法!"; 
}

    # 八、单链表结点的删除

    # 时间复杂度:O(n)

     //删除第i个结点元素  
void ListDel(List *L,int i){
	int j=0;						//计数器
	 while(L->next&&j<i-1){			//注意此处判断条件为L->next,而不是L!!!!
		L=L->next;
		j++;
	 }
	if(L->next&&j==i-1){			//注意此处判断条件为L->next,而不是L!!!!
	List *p=L->next;				//需要临时空间储存要删除的结点 
	L->next=L->next->next;			//关键步骤 
    delete p;
	 } 
	 else cout<<"删除位置非法!"; 
}
