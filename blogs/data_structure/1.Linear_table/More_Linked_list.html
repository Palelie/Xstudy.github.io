    双向链表与循环链表

    默 12/9/2021 数据结构

    # 双向链表、循环链表

    # 一、循环链表

    # 1、循环链表：头尾相接的链表(即表中最后一个结点的指针域指向头结点，整个链表形成一个环)

    # 注意：循环链表一定要有头结点

    # 2、优点：从表中任一结点出发均可访问全部结点

    # 3、与单链表的主要差异：链表遍历时，判别当前指针p是否指向表尾结点的终止条件不同

    • 单链表：p!=NULL或 p->next!=NULL
    • 循环链表： p!=L 或 p->next!=L

    # 4、头指针与尾指针表示的循环链表的区别

    # 建议用尾指针来表示循环链表，特别是需要经常操作首尾元素的循环链表，用尾指针表示效率高很多（注意此处说的是尾指针不是尾结点！）

    # 5、带尾指针的循环链表的合并

    # 完整代码

    #include<iostream>
using namespace std;

//链表结点 
struct List{
	int data;
	List* next;
};

//尾插法创建用尾指针表示的循环链表 
void CreateList(List*&L,int *l,int n){
	//L为尾指针
	L=new List;						//头结点
	L->next=L; 						//构建循环 
	for(int i=0;i<n;i++){
		List *p=new List;			
		p->data=l[i];
		p->next=L->next;			//尾插法：新结点指向头结点 
		L->next=p;					
		L=p; 						//尾指针指向新加入的结点 
	}
}

//输出链表
void  PutList(List *L){
	List *p=L->next->next;			//让p指向首元结点 
	while(p!=L->next){				//循环链表的判断条件 
		cout<<p->data<<' ';
		p=p->next;
	}
}

//两个用尾指针表示的循环链表的合并
List* LinkCombine(List *L1,List *L2){
	List *p=L1->next;				//p存放L1头结点 
	L1->next=L2->next->next;		//L1表尾与L2首元结点连接 
	delete L2->next;				//删除L2头结点 
	L2->next=p;						//L2为新链表的尾指针 
	return L2;						//返回新链表的尾指针 
} 

int main(){
    int l1[5]={5,9,4,35,15};		//测试数组1 
    int l2[6]={7,63,39,41,21,7}; 	//测试数组2 
	List *t1,*t2;					//创建两个尾指针
	CreateList(t1,l1,5);			//创建循环链表 
	CreateList(t2,l2,6);
	cout<<"链表1：";
	PutList(t1);
	cout<<endl<<"链表2：";
	PutList(t2);
	t1=LinkCombine(t1,t2);			//将L1和L2链表合并，并将新链表的尾指针赋给t1
	cout<<endl<<"合并后的链表：";
	PutList(t1);
	return 0;
}

    # 二、双向链表

    # 1、为什么要有双向链表？

    ​

    # 为了方便查找前驱结点

    # 故不需要尾指针

    # 2、双向循环链表

    # 步骤：

    1. 让头结点的前驱指针指向链表的最后一个结点
    2. 最后一个结点的后继指针指向头结点

    #

    # 3、双向链表的插入

    # 单链表只需修改两个指针，双向链表则需要修改四个指针，所以关键步骤有四步。时间复杂度O(n)

    # 4、双向链表的删除

    # 关键步骤有两步

    # 完整代码

    #include<iostream>
using namespace std;

struct List{
	int data;
	List *prior; 
	List *next;
};

//头插法创建双向循环链表 
void CreateList(List*&L,int *l,int n){
	L=new List;					//创建头结点 
	L->next=L;					//初始化双向循环链表 
	L->prior=L;
	for(int i=0;i<n;i++){
		List *p=new List;
		p->data=l[i];
		L->next->prior=p;		//插入结点关键步骤 
		p->next=L->next;		//插入结点关键步骤 
		L->next=p;				//插入结点关键步骤 
		p->prior=L;				//插入结点关键步骤  
	}
}

//输出双向链表
void PutList(List *L){
	List *p=L->next;
	while(p!=L){
		cout<<p->data<<' ';
		p=p->next;
	}
} 

//在第i位前插入数字e 
void ListInsert(List *L,int i,int e){
	int j;							//计数器 
	List *p=L->next;				//从首元结点开始数 
    
	//p!=L是为了防止i超过链表长度，所以无法从头结点开始数，所以无法找到第i-1个元素并在该元素后插入
	//（默认已经从第一个开始数，无法插入到第一个之前） 
	for(j=1;j<i&&p!=L;j++){			 
		p=p->next;
	}
	if(j==i){
		List *q=new List;
		q->data=e;
		p->prior->next=q;			//关键步骤 
		q->prior=p->prior;			//关键步骤 
		q->next=p;					//关键步骤 
		p->prior=q;					//关键步骤 
	}else cout<<"插入位置非法！";
}

//删除第i位元素
void ListDelete(List *L,int i){
	int j=1;
	List *p=L->next;
	while(j<i&&p!=L){				//定位删除结点位置方法同插入算法 
		p=p->next;
		j++;
	}
	if(j==i&&p!=L){					//不能删除头结点 
		p->prior->next=p->next;
		p->next->prior=p->prior;
		delete p;
	} else cout<<"删除位置非法！";
} 

int main(){
	int l[5]={5,9,4,35,15};			//实验数组1 
	int i=6,e=10;					//i为要操作的链表位置，e为插入元素 
	List *head;
    
	//创建链表 
	CreateList(head,l,5);
	cout<<"循环链表："; 
	PutList(head);
    
	//链表元素插入 
	ListInsert(head,i,e);
	cout<<endl<<"在第"<<i<<"位前插入数字"<<e<<':';
	PutList(head);
    
	//链表元素插入 
	ListDelete(head,i);
	cout<<endl<<"删除第"<<i<<"位数字";
	PutList(head);
	return 0;
}

    # 三、小结

    # 1、链式存储结构的优点

    • 结点空间可以动态申请和释放
    • 数据元素的逻辑次序靠结点的指针来指示，插入和删除不需要移动元素

    # 2、链式存储结构的缺点

    • 链式存储结构是（顺序存取结构）非随机存取结构。对任一结点的操作都要从头指针依指针链查找到该结点，这增加了算法的复杂度。(对某个结点操作一般要先找到该结点)
    • 存储密度小，每个结点的指针域需额外占用存储空间。当每个结点的数据域所占的字节数不多时，指针域所占的存储空间的比重显得很大
    • 存储密度是指结点数据本身占用的空间/结点占用的空间总量
