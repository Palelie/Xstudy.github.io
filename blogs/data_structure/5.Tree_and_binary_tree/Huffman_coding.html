    # 步骤

    1. 构造哈夫曼树
    2. 从叶子结点（1~n）开始，回溯到根结点（parent！=0）
    3. 由路径得到编码，若子节点为父节点的左孩子记0，右孩子记1
    4. 将0和1依次存入临时数组，最后将数组反向输出

    #include<iostream>
using namespace std;

struct hufftree{
	int elem,par,lc,rc;
};

void huff(hufftree *t,int n){
	int term[4];
	for(int i=1;i<=n-1;i++){						//若有n个树，则需合并n-1次 
		for(int k=0;k<2;k++) term[k]=9999999;		
        //每次都进行初始化，0记录最小值，1记录第二小的值，2记录最小值位置，3记录第二小值的位置
        
        
		//搜索最小数及次小数（版本一：时间复杂度n*n，pass） 
//		for(int f=0;f<2;f++){						//进行两次搜最小值 
//			for(int j=1;j<=n-1+i;j++){				//从原始数据遍历到最后一个加入的结点 
//				if(!t[j].par&&t[j].elem<term[f]&&j!=term[2]){		
        //搜最小值。条件：1、没有父节点；2、第二遍搜索需排除第一遍搜出的最小值 
//				term[f]=t[j].elem;					//记录数值 
//				term[f+2]=j;						//记录位置 
//				}
//			}
//		}
        
        
		//搜索最小数及次小数（版本二：时间复杂度n）
		for(int f=1;f<=n+i-1;f++){
			if(!t[f].par){							//条件：没有父节点 
				if(t[f].elem<term[0]){
					term[1]=term[0];
					term[3]=term[2];
					term[0]=t[f].elem;
					term[2]=f;
				}
				else if(t[f].elem<term[1]){
					term[1]=t[f].elem;
					term[3]=f;
				}
			}
		} 
		t[n+i].elem=term[0]+term[1];				//记录新结点的值 
		t[n+i].lc=term[2];							//新结点的左孩子 
		t[n+i].rc=term[3];							//新结点的右孩子 
		t[term[2]].par=n+i;							//给两个被合并的子叶添加父节点位置信息 
		t[term[3]].par=n+i;
	}
}

void cou(hufftree *t,int n){			//构造并输出哈夫曼编码 
	int term[n],sum=0;					//储存每个数字的哈夫曼编码 
	float len=0;						//计算长度（高度） 
	for(int i=1;i<=n;i++){				//输出n位数字的哈夫曼编码 
		int x=i;
		int p=t[x].par,k=0;
		while(p){						//构造哈夫曼编码 
			if(t[p].lc==x) term[k++]=0;
			else term[k++]=1;
			x=p;
			p=t[p].par;
		}
		sum+=t[i].elem;					
		len+=k*t[i].elem;				//计算带权路径长度 
		cout<<t[i].elem<<":	";
		for(int j=k-1;j>=0;j--) cout<<term[j];
		cout<<endl;
	}
	float l=len/sum;
	cout<<endl<<"平均长度="<<l;
}
 
int main(){
	int n;
	cout<<"请输入数据个数："; 
	cin>>n;
	hufftree *t=new hufftree[2*n+1];		//创建哈夫曼表格 
	for(int i=1;i<=2*n;i++){				//初始化哈夫曼表格 
		t[i].elem=0;
		t[i].par=0;
		t[i].lc=0;
		t[i].rc=0;
	}
	cout<<"请输入数据：";						//测试数据：11 4 6 2 13 25 8 16 
	for(int i=1;i<=n;i++) cin>>t[i].elem;	//输入初始数据 
	huff(t,n);								//创建哈夫曼树 
	cou(t,n);								//输出哈夫曼编码 
	return 0;
}

    # 教师c语言版

    #include <stdio.h>
#include <malloc.h>
#include <string.h>
#define N 50 //叶子结点数目的最大数
#define M 2*N-1 //树中结点总数。完全二叉树中n2=n0-1,n=n0+n2=2n0-1 
typedef struct{
	char data[5];//结点值 
	int weight;//权重 
	int parent;
	int lchild;
	int rchild;
}HTNode;//定义哈夫曼树存储结构
 
typedef char **HuffmanCode;//定义哈夫曼编码存储结构 

//构造具有n个结点的哈夫曼树
void CreateHT(HTNode ht[], int n){
	char *str[]={"11","4","6","2","13","25","8","16"};//待编码字符，这是一个字符串数组
	int fnum[]={11,4,6,2,13,25,8,16};//待编码字符出现的频率
	int i,k,lnode,rnode;
	int min1,min2;
	
	for(i=0;i<n;i++){
		//用待编码字符以及频率初始化ht[]数组
		strcpy(ht[i].data,str[i]);//strcpy将str[]复制到ht[]中 
		ht[i].weight=fnum[i]; 
	} 
	
	for(i=0;i<2*n-1;i++){
		//n个叶子结点，总共2n-1个结点 
		//ht[]数组中所有结点的相关域初值设置为-1 
		ht[i].parent=-1;
		ht[i].lchild=-1;
		ht[i].rchild=-1;
	} 
	
	for(i=n;i<2*n-1;i++){
		min1=32767;min2=32767;//存放最小和次小权重值
		lnode=-1;rnode=-1;//最小和次小权重的结点位置
		
		for(k=0;k<=i-1;k++){//循环遍历寻找权值最小和次小的结点 
			if(ht[k].parent==-1){
			//根据parent是否等于-1来判断此结点是否已经跟另一结点结合成新结点。
			//那么循环结束后需要更新此轮所选出的最小和次小结点的parent值。 
				if(ht[k].weight<min1){
					min2=min1;
					rnode=lnode;
					min1=ht[k].weight;
					lnode=k;
				}
				else{
					if(ht[k].weight<min2){
						min2=ht[k].weight;
						rnode=k;
					}
				}		
			}	 
		}
		
		//设置双亲结点和权值 
		ht[rnode].parent=i;
		ht[lnode].parent=i;//最小和次小权重结点的双亲
		ht[i].weight=ht[rnode].weight+ht[lnode].weight;// 最小和次小权重结点双亲的权值
		
		//设置双亲结点的lchild和rchild。位序小的放在左子树上，位序大的放在右子树上 
		if(lnode>rnode){
			ht[i].lchild=lnode;
			ht[i].rchild=rnode;
		}
		else{
			ht[i].lchild=rnode;
			ht[i].rchild=lnode;
		} 	
	}
}

//构造哈夫曼编码hcd 
void CreateHCode(HTNode ht[], HuffmanCode hcd, int n){
	int i,f,c,temp;
	char *hc;//存放哈夫曼编码的工作数组
	
	//根据哈夫曼树求n个叶子的哈夫曼编码 
	for(i=0;i<n;i++){
		hc=(char *)malloc(n*sizeof(char));//分配存放编码的工作区间 
		hc[n]='\0';
		temp=n-1;//temp指示编码字符在hc中首位置
		c=i;//c为被编码字符串位置
		f=ht[i].parent;//f为c的双亲位置
		while(f!=-1){
			if(ht[f].lchild==c)
				hc[temp--]='0';//左孩子的编码 
			else
				hc[temp--]='1';//右孩子的编码
			c=f;
			f=ht[f].parent;  
		} 
		hcd[i]=(char*)malloc((n-temp)*sizeof(char));//给第i个字符编码分配空间
		strcpy(hcd[i],hc+temp+1);
	} 
}

//输出哈夫曼编码 
void DispHCode(HTNode ht[], HuffmanCode hcd, int n){
	int i;
	int sum=0,m=0,j;
	printf("输出哈夫曼编码:\n");
	for(i=0;i<n;i++){
		j=0;
		printf("%s:\t",ht[i].data);
		while(hcd[i][j]!='\0'){
			printf("%c",hcd[i][j]);
			j++;
		}
		m+=ht[i].weight;//计算码长，求路径长。
		sum+=ht[i].weight*j;
		printf("\n"); 
	}
	printf("\n平均长度=%g\n",1.0*sum/m);
}

int main(){
	int n=8;
	HTNode ht[M];
	HuffmanCode hcd;
	hcd=(HuffmanCode)malloc((2*n)*sizeof(char*));
	CreateHT(ht,n);
	CreateHCode(ht,hcd,n);
	DispHCode(ht,hcd,n);
	printf("\n");
	return 0;
}
