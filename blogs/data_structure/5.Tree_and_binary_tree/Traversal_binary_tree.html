    # 1、非递归前序遍历

    # 完整代码

    #include<iostream>
#include<stack>
using namespace std;

struct priBitree{
	char data;
	priBitree *lc;
	priBitree *rc;
};

//按照前序递归遍历建立二叉树
void creatBitree(priBitree*&p){
	char ch;
	cin>>ch;
	if(ch=='#') p=NULL;
		else {
			p=new(priBitree);
			p->data=ch;
			creatBitree(p->lc);
			creatBitree(p->rc);
		}
}

//前序非递归遍历 
void show(priBitree *p){
		stack<priBitree*>s;					//创建一个c++的栈容器s 
		while(p!=NULL||!s.empty()){			//判断整棵树是否遍历完成 
			while(p!=NULL){					//当结点不为空 
				s.push(p);					//将结点压入栈中（此时为第一次经过该点）
				cout<<p->data;				//输出该点  
				p=p->lc;					//遍历左子树 
			}
			if(!s.empty()){					//结点为空时，若栈中还有元素，说明此时p为某叶子结点的左空结点 
				p=s.top();					//此时为第二次经过该点 
				s.pop();					//弹栈 
				p=p->rc;					//遍历右子树 
			}
		}
}

int main(){
	cout<<"please input priBitree:";
	priBitree *h;
	creatBitree(h);//764##2#3##1#30##1##（测试数据）
	show(h);
	return 0;
}

    # 3、非递归的前序和中序遍历对比

    //前序非递归遍历 
void show(priBitree *p){
		stack<priBitree*>s;					
		while(p!=NULL||!s.empty()){			
			while(p!=NULL){					
				s.push(p);					//将结点压入栈中（此时为第一次经过该点）
				cout<<p->data;				//输出该点（入栈时输出该点！）
				p=p->lc;					
			}
			if(!s.empty()){					
				p=s.top();					//此时为第二次经过该点 
				s.pop();					
				p=p->rc;					
			}
		}
}

//中序非递归遍历 
void show(priBitree *p){
		stack<priBitree*>s;					
		while(p!=NULL||!s.empty()){			
			while(p!=NULL){					
				s.push(p);					//将结点压入栈中 （此时为第一次经过该点）
				p=p->lc;					
			}
			if(!s.empty()){					
				p=s.top();					//此时为第二次经过该点 
				s.pop();					 
				cout<<p->data;				//输出该点（出栈时输出该点）
				p=p->rc;					
			}
		}
}

    # 总结：前序和中序的非递归算法仅在输出结点的位置不同。

    # 原因：入栈都是按照根—左—右的顺序

    # 先序遍历入栈输出：根（先入栈先输出）—左（后入栈后输出）—右

    # 中序遍历出栈输出：根（后出栈后输出）—左（先出栈先输出）—右

    # 4、非递归后序遍历

    # 完整代码

    #include<iostream>
#include<stack>
using namespace std;

struct priBitree{
	char data;
	priBitree *lc;
	priBitree *rc;
};

struct term{					//在保持原二叉树结构不变的情况下进行后序非递归遍历，需要借助新的储存单位 
	int flag;					//记录当前结点入栈次数 
	priBitree *t;				//与当前结点绑定 
};

//按照前序递归遍历建立二叉树
void creatBitree(priBitree*&p){
	char ch;
	cin>>ch;
	if(ch=='#') p=NULL;
		else {
			p=new(priBitree);
			p->data=ch;
			creatBitree(p->lc);
			creatBitree(p->rc);
		}
}

//非递归后序遍历 
void show(priBitree *p){
		stack<term*>s;						//栈的类型需改为借助的新存储结构 
		term *ter;
		while(p!=NULL||!s.empty()){			//判断整棵树是否已经遍历完成 
			if(p!=NULL){					//当结点不为空——入栈并访问左子树 
				ter=new(term); 				//赋予新结构空间 
				ter->t=p;					//指向（绑定）该结点 ，此时ter->t即为原结点 
				ter->flag=1;				//记录该结点入栈次数 
				s.push(ter);				//将结点入栈 
				p=p->lc;					//遍历左子树 
			}
			else{							//当结点为空 
				ter=s.top();				//先出栈 
				p=ter->t;
				s.pop();
				if(ter->flag==1){			//判断出栈的这个结点已经入栈几次，若为一次则说明右子树还未遍历，需要再入栈 
					s.push(ter);			//入栈 
					ter->flag=2;			//入栈后标记为已经入栈两次 
					p=p->rc;				//遍历右子树 
				}
				else{						//该条件为flag=2，说明出栈的这个结点已经入栈过两次，可以输出了 
					cout<<p->data;			//输出该结点 
					p=NULL;					//将p改为空，以便下次循环继续出栈（因为这棵子树已经完全遍历完成） 
				}
			}
			
		}
	
}

int main(){
	cout<<"please input priBitree:";
	priBitree *h;
	creatBitree(h);//764##2#3##1#30##1##（测试数据）
	show(h);
	return 0;
}
